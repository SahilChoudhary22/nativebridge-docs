## Quick Example

See how easy it is to communicate between JavaScript and native code:

<CodeGroup>

```javascript JavaScript
// Call native method from JavaScript
const result = await NativeBridge.execute('getUserLocation');
console.log('User location:', result);

// Register a handler for native events
NativeBridge.on('locationUpdate', (data) => {
  console.log('New location:', data);
});

// Call native method with parameters
const photo = await NativeBridge.execute('takePhoto', {
  quality: 'high',
  camera: 'front'
});
```

```swift iOS (Swift)
// Register native method
NativeBridge.register("getUserLocation") { params, callback in
    let location = LocationManager.shared.currentLocation
    callback(.success([
        "latitude": location.latitude,
        "longitude": location.longitude
    ]))
}

// Emit event to JavaScript
NativeBridge.emit("locationUpdate", data: [
    "latitude": newLocation.latitude,
    "longitude": newLocation.longitude
])
```

```kotlin Android (Kotlin)
// Register native method
NativeBridge.register("getUserLocation") { params, callback ->
    val location = LocationManager.getCurrentLocation()
    callback.success(mapOf(
        "latitude" to location.latitude,
        "longitude" to location.longitude
    ))
}

// Emit event to JavaScript
NativeBridge.emit("locationUpdate", mapOf(
    "latitude" to newLocation.latitude,
    "longitude" to newLocation.longitude
))
```

</CodeGroup>

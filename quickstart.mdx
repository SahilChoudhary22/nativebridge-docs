## Overview

This guide will walk you through setting up NativeBridge in your hybrid mobile application. By the end, you'll have a working bridge between your JavaScript code and native iOS/Android functionality.

## Prerequisites

Before you begin, make sure you have:

  <Accordion icon="check" title="Development Environment">

  <Accordion icon="code" title="Basic Knowledge">
    - JavaScript/TypeScript fundamentals
    - Basic understanding of iOS (Swift) or Android (Kotlin/Java) development

## Step 1: Initialize NativeBridge

### JavaScript Setup

Create a bridge instance in your JavaScript application:

```javascript bridge.js
import { NativeBridge } from '@nativebridge/core';

// Initialize the bridge
const bridge = new NativeBridge({
  debug: true, // Enable debug logging
  timeout: 5000, // Default timeout for native calls
});

// Export for use throughout your app
export default bridge;
```

## Step 2: Configure Native Platforms

<Tabs>
  <Tab title="iOS (Swift)">
    ```swift AppDelegate.swift
    import UIKit
    import NativeBridge

    @main
    class AppDelegate: UIResponder, UIApplicationDelegate {

        func application(_ application: UIApplication,
                        didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {

            // Initialize NativeBridge
            NativeBridge.shared.initialize()

            // Register your modules
            NativeBridge.shared.registerModule(DeviceModule())
            NativeBridge.shared.registerModule(StorageModule())

            return true
        }
    }
    ```
  </Tab>
  <Tab title="Android (Kotlin)">
    ```kotlin MainActivity.kt
    import android.os.Bundle
    import androidx.appcompat.app.AppCompatActivity
    import com.nativebridge.NativeBridge

    class MainActivity : AppCompatActivity() {

        override fun onCreate(savedInstanceState: Bundle?) {
            super.onCreate(savedInstanceState)

            // Initialize NativeBridge
            NativeBridge.initialize(this)

            // Register your modules
            NativeBridge.registerModule(DeviceModule())
            NativeBridge.registerModule(StorageModule())

            setContentView(R.layout.activity_main)
        }
    }
    ```
  </Tab>
</Tabs>

## Step 3: Create Your First Bridge Module

Let's create a simple module that provides device information:

### Native Implementation

<Tabs>
  <Tab title="iOS (Swift)">
    ```swift DeviceModule.swift
    import Foundation
    import UIKit
    import NativeBridge

    class DeviceModule: NativeBridgeModule {

        override func getName() -> String {
            return "Device"
        }

        @BridgeMethod
        func getDeviceInfo(_ params: [String: Any],
                          callback: @escaping BridgeCallback) {
            let info = [
                "model": UIDevice.current.model,
                "systemName": UIDevice.current.systemName,
                "systemVersion": UIDevice.current.systemVersion,
                "identifier": UIDevice.current.identifierForVendor?.uuidString ?? ""
            ]

            callback(.success(info))
        }

        @BridgeMethod
        func vibrate(_ params: [String: Any],
                    callback: @escaping BridgeCallback) {
            AudioServicesPlaySystemSound(kSystemSoundID_Vibrate)
            callback(.success(nil))
        }
    }
    ```
  </Tab>
  <Tab title="Android (Kotlin)">
    ```kotlin DeviceModule.kt
    import android.content.Context
    import android.os.Build
    import android.os.VibrationEffect
    import android.os.Vibrator
    import com.nativebridge.BridgeModule
    import com.nativebridge.BridgeMethod
    import com.nativebridge.BridgeCallback

    class DeviceModule(private val context: Context) : BridgeModule() {

        override fun getName(): String = "Device"

        @BridgeMethod
        fun getDeviceInfo(params: Map<String, Any>, callback: BridgeCallback) {
            val info = mapOf(
                "model" to Build.MODEL,
                "manufacturer" to Build.MANUFACTURER,
                "systemVersion" to Build.VERSION.RELEASE,
                "identifier" to getDeviceId()
            )

            callback.success(info)
        }

        @BridgeMethod
        fun vibrate(params: Map<String, Any>, callback: BridgeCallback) {
            val vibrator = context.getSystemService(Context.VIBRATOR_SERVICE) as Vibrator

            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                vibrator.vibrate(VibrationEffect.createOneShot(200, VibrationEffect.DEFAULT_AMPLITUDE))
            } else {
                vibrator.vibrate(200)
            }

            callback.success(null)
        }
    }
    ```
  </Tab>
</Tabs>

### JavaScript Usage

Now you can use your native module from JavaScript:

```javascript app.js
import bridge from './bridge';

// Get device information
async function getDeviceInfo() {
  try {
    const info = await bridge.execute('Device.getDeviceInfo');
    console.log('Device Info:', info);
    // Output: { model: "iPhone", systemName: "iOS", ... }
  } catch (error) {
    console.error('Failed to get device info:', error);
  }
}

// Trigger device vibration
async function vibrateDevice() {
  try {
    await bridge.execute('Device.vibrate');
    console.log('Vibration triggered');
  } catch (error) {
    console.error('Failed to vibrate:', error);
  }
}

// Use the functions
getDeviceInfo();
vibrateDevice();
```

## Step 4: Handle Events

NativeBridge supports bidirectional communication. Here's how to handle events from native code:

### Emit from Native

<Tabs>
  <Tab title="iOS (Swift)">
    ```swift
    // Emit an event to JavaScript
    NativeBridge.shared.emit("batteryLow", data: [
        "level": 10,
        "isCharging": false
    ])
    ```
  </Tab>
  <Tab title="Android (Kotlin)">
    ```kotlin
    // Emit an event to JavaScript
    NativeBridge.emit("batteryLow", mapOf(
        "level" to 10,
        "isCharging" to false
    ))
    ```
  </Tab>
</Tabs>

### Listen in JavaScript

```javascript
// Subscribe to native events
const unsubscribe = bridge.on('batteryLow', (data) => {
  console.log('Battery is low:', data.level + '%');
  if (!data.isCharging) {
    alert('Please charge your device!');
  }
});

// Unsubscribe when no longer needed
unsubscribe();
```

## Step 5: Error Handling

Implement proper error handling for robust communication:

```javascript
// Wrap native calls with error handling
async function safeNativeCall(method, params = {}) {
  try {
    const result = await bridge.execute(method, params);
    return { success: true, data: result };
  } catch (error) {
    console.error(`Native call failed: ${method}`, error);

    // Handle specific error types
    if (error.code === 'MODULE_NOT_FOUND') {
      console.error('Module not registered on native side');
    } else if (error.code === 'METHOD_NOT_FOUND') {
      console.error('Method does not exist in native module');
    } else if (error.code === 'TIMEOUT') {
      console.error('Native call timed out');
    }

    return { success: false, error: error.message };
  }
}

// Usage
const result = await safeNativeCall('Device.getDeviceInfo');
if (result.success) {
  console.log('Device info:', result.data);
} else {
  console.error('Failed to get device info:', result.error);
}
```
